            >
                <Button variant="outlined" onClick={onClose}>
                    Cancel
                </Button>
                <Button 
                    variant="outlined"
                    onClick={handleApply}
                    disabled={isAgentNode && !selectedAgentId}
                >
                    Apply
                </Button>
                <Button 
                    variant="contained" 
                    onClick={handleSave}
                    disabled={isAgentNode && !selectedAgentId}
                >
                    Save & Close
                </Button>
            </Box>
        </Box>
    )
}

export default NodeConfigPanel
